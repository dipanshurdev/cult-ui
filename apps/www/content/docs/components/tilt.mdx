## Usage

```tsx
import { Tilt } from "@/components/ui/tilt"
```

```tsx
export default function TiltDemo() {
  return (
    <Tilt
      className="w-72 h-96 rounded-3xl bg-gradient-to-br from-indigo-500 to-purple-500 shadow-2xl flex flex-col items-center justify-center p-8 text-white"
      maxRotation={20}
      scale={1.1}
    >
      <div style={{ transform: "translateZ(50px)" }}>
        <h3 className="text-2xl font-bold">Tilt Me</h3>
      </div>
    </Tilt>
  )
}
```

## Props

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `maxRotation` | `number` | `15` | The maximum rotation in degrees. |
| `scale` | `number` | `1.05` | The scale factor on hover. |
| `springOptions` | `SpringOptions` | `{ stiffness: 300, damping: 30, mass: 1 }` | Spring animation configuration. |
